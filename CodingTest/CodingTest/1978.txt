#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std;

int main()
{
	int cnt = 0;
	int check_cnt = 0;
	int num = 0, input;


	cin >> num;

	for(int i = 0; i < num; i++)
	{
		cin >> input;

		for(int j = 1; j <= input; j++)
		{
			if (input % j == 0)
				cnt++;
		}
		if (cnt == 2)
			check_cnt++;
		cnt = 0;
	}

	cout << check_cnt << endl;

	return 0;
}
